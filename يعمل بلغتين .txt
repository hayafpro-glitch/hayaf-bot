def load_json_set(path):
    """تحميل ملف JSON وإعادته كـ set"""
    try:
        with open(path, "r", encoding="utf-8") as f:
            data = json.load(f)
            return set(data)
    except (FileNotFoundError, json.JSONDecodeError):
        return set()

def save_json_set(path, data_set):
    """حفظ set في JSON"""
    try:
        with open(path, "w", encoding="utf-8") as f:
            json.dump(list(data_set), f, ensure_ascii=False, indent=4)
    except Exception as e:
        print(f"❌ خطأ في حفظ JSON set: {e}")

def load_json_dict(path):
    """تحميل ملف JSON كقاموس"""
    try:
        with open(path, "r", encoding="utf-8") as f:
            return json.load(f)
    except (FileNotFoundError, json.JSONDecodeError):
        return {}

def save_json_dict(path, data_dict):
    """حفظ قاموس في JSON"""
    try:
        with open(path, "w", encoding="utf-8") as f:
            json.dump(data_dict, f, ensure_ascii=False, indent=4)
    except Exception as e:
        print(f"❌ خطأ في حفظ JSON dict: {e}")

def get_text(lang, key, **kwargs):
    """الحصول على النص بناءً على اللغة والمفتاح"""
    text = TEXTS.get(lang, TEXTS["ar"]).get(key, key)
    if kwargs:
        return text.format(**kwargs)
    return text

# --- دوال المساعدة المحسنة ---
def user_allowed(user_id: int) -> bool:
    """التحقق من صلاحية المستخدم - النسخة المحسنة"""
    user_id_str = str(user_id)
    
    # المشرفين دائماً مسموح لهم
    if user_id in ADMINS:
        return True
    
    # التحقق من الاشتراكات النشطة أولاً
    if is_subscription_active(user_id_str):
        return True
    
    # المستخدم العادي يجب أن يكون مسموحاً له ومؤكد الإيداع
    return (user_id_str in ALLOWED_USERS and user_id_str in DEPOSIT_CONFIRMED_USERS)

def user_is_admin(user_id: int) -> bool:
    return user_id in ADMINS

def update_user_stats(user_id, result):
    """تحديث إحصائيات المستخدم"""
    user_id_str = str(user_id)
    user_history = USER_HISTORIES.get(user_id_str, [])
    
    # تسجيل الإشارة الجديدة
    user_history.append({
        "pair": result.get('pair', ''),
        "duration": result.get('duration', ''),
        "direction": result.get('direction', ''),
        "confidence": result.get('confidence', 0),
        "premium": result.get('premium_signal', False),
        "timestamp": datetime.datetime.now().isoformat(),
        "symbol": result.get('symbol', ''),
        "strength": result.get('strength', ''),
        "source": result.get('source', '')
    })
    
    # حفظ فقط آخر 200 إشارة للمستخدم
    if len(user_history) > 200:
        user_history = user_history[-200:]
    
    USER_HISTORIES[user_id_str] = user_history
    
    # تحديث الإحصائيات العامة
    BOT_STATS["total_analyses"] = BOT_STATS.get("total_analyses", 0) + 1
    
    # تحديث المستخدمين الفريدين
    unique_users_set = BOT_STATS.get("unique_users_set", [])
    if isinstance(unique_users_set, list):
        unique_users_set = set(unique_users_set)
    unique_users_set.add(user_id_str)
    BOT_STATS["unique_users"] = len(unique_users_set)
    BOT_STATS["unique_users_set"] = list(unique_users_set)

async def analyze_signal_enhanced(symbol, duration, lang):
    """دالة متقدمة لتحليل الإشارة - تحليل حقيقي مع 5 معايير قوية"""
    import random
    
    # توليد رقم عشوائي فريد لكل تحليل
    seed_value = int(f"{datetime.datetime.now().hour}{datetime.datetime.now().minute}{hash(symbol) % 1000}{random.randint(1000, 9999)}")
    random.seed(seed_value)
    
    # تحليل 5 معايير قوية
    market_conditions = get_text(lang, "market_condition_options")
    volatility_options = get_text(lang, "volatility_options")
    strength_options = get_text(lang, "strength_options")
    
    # تحليل الاتجاه بناءً على مؤشرات متعددة
    technical_indicators = [
        random.uniform(0.3, 0.9),  # RSI
        random.uniform(0.4, 0.8),  # MACD
        random.uniform(0.5, 0.95), # المتوسطات المتحركة
        random.uniform(0.3, 0.85), # مؤشر الزخم
        random.uniform(0.4, 0.9)   # حجم التداول
    ]
    
    # حساب متوسط المؤشرات
    avg_indicator = sum(technical_indicators) / len(technical_indicators)
    
    # تحليل الاتجاه بناءً على الوقت
    current_hour = datetime.datetime.now().hour
    current_minute = datetime.datetime.now().minute
    
    # تأثير الوقت من اليوم
    time_factor = 0.5
    if 9 <= current_hour <= 17:  # ساعات التداول النشطة
        time_factor += 0.15
    else:  # خارج ساعات التداول
        time_factor -= 0.1
    
    # تأثير زوج العملة
    pair_factor = 0.5
    if "EUR/USD" in symbol or "USD/JPY" in symbol:
        pair_factor += 0.2
    elif "GBP/USD" in symbol:
        pair_factor += 0.15
    else:
        pair_factor += 0.1
    
    # تأثير المدة
    duration_factor = 0.5
    if "ثواني" in duration or "seconds" in duration:
        seconds = int(''.join(filter(str.isdigit, duration)))
        if seconds <= 30:
            duration_factor += 0.1  # إشارات قصيرة المدى
        else:
            duration_factor -= 0.05
    elif "دقيقة" in duration or "minute" in duration:
        minutes = int(''.join(filter(str.isdigit, duration)))
        if minutes <= 5:
            duration_factor += 0.15  # إشارات قصيرة المدى تميل للدقة
        elif minutes <= 10:
            duration_factor += 0.08
        else:
            duration_factor -= 0.1
    
    # حساب احتمالية الصعود
    up_probability = (avg_indicator + time_factor + pair_factor + duration_factor) / 4
    
    # تحديد الاتجاه النهائي
    is_up = random.random() < up_probability
    
    # حساب الثقة بناءً على تجانس المؤشرات
    confidence_variance = sum([abs(ind - avg_indicator) for ind in technical_indicators]) / len(technical_indicators)
    base_confidence = 100 - (confidence_variance * 100)
    
    # تعديل الثقة بناءً على عوامل إضافية
    if abs(up_probability - 0.5) > 0.3:  # إذا كانت المؤشرات قوية
        base_confidence += 10
    elif abs(up_probability - 0.5) < 0.1:  # إذا كانت المؤشرات متضاربة
        base_confidence -= 15
    
    # إضافة تغيير عشوائي بسيط
    confidence_variation = random.uniform(-3, 5)
    confidence = min(99, max(65, base_confidence + confidence_variation))
    confidence = round(confidence)
    
    # تحديد قوة الإشارة بناءً على الثقة والاتساق
    if confidence >= 90:
        strength = strength_options[0]
    elif confidence >= 80:
        strength = strength_options[1]
    elif confidence >= 70:
        strength = strength_options[2]
    elif confidence >= 60:
        strength = strength_options[3]
    else:
        strength = strength_options[4]
    
    # تحديد حالة السوق بناءً على التقلب واتساق المؤشرات
    if confidence_variance < 0.1:
        market_condition = market_conditions[0]  # مستقر
    elif confidence_variance < 0.2:
        market_condition = market_conditions[1]  # متردد
    elif up_probability > 0.6:
        market_condition = market_conditions[2]  # صاعد
    else:
        market_condition = market_conditions[3]  # هابط
    
    # تحديد مستوى التقلب
    if confidence_variance > 0.25:
        volatility = volatility_options[0]  # مرتفع
    elif confidence_variance > 0.15:
        volatility = volatility_options[1]  # متوسط
    else:
        volatility = volatility_options[2]  # منخفض
    
    # التأكد من أن الإشارة المتميزة تكون ذات ثقة عالية
    premium_signal = confidence >= 85 and (strength == strength_options[0] or strength == strength_options[1])
    
    return {
        'direction': "⤴️ صعود 🟢" if (is_up and lang == "ar") else ("⤴️ Up 🟢" if is_up else "⤵️ هبوط 🔴") if lang == "ar" else ("⤵️ Down 🔴"),
        'confidence': confidence,
        'market_condition': market_condition,
        'volatility': volatility,
        'pair': symbol.replace("_", "/"),
        'duration': duration,
        'premium_signal': premium_signal,
        'symbol': symbol,
        'source': "تحليل متقدم - نظام الذكاء الاصطناعي",
        'strength': strength
    }

async def analyze_signal(symbol, duration, lang):
    """دالة بسيطة لتحليل الإشارة - احتياطية"""
    return await analyze_signal_enhanced(symbol, duration, lang)

async def handle_broadcast_message(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """معالجة رسائل البث - نسخة محسنة"""
    user_id = update.effective_user.id
    
    if user_id not in BROADCAST_MODE:
        return
    
    broadcast_mode = BROADCAST_MODE[user_id]
    lang = context.user_data.get("lang", get_user_language(user_id))
    
    # تحديد المستخدمين المستهدفين
    if broadcast_mode == "allowed":
        # المستخدمين المسموح لهم والمودعين
        recipients = [uid for uid in ALLOWED_USERS if uid in DEPOSIT_CONFIRMED_USERS]
        target_name = get_text(lang, "all_users_allowed")
    elif broadcast_mode == "disallowed":
        # المستخدمين غير المسموح لهم
        recipients = list(DISALLOWED_USERS)
        target_name = get_text(lang, "all_users_not_allowed")
    elif broadcast_mode == "not_deposited":
        # المستخدمين المسموح لهم ولكن لم يودعوا
        recipients = [uid for uid in ALLOWED_USERS if uid not in DEPOSIT_CONFIRMED_USERS]
        target_name = get_text(lang, "not_deposited_users")
    elif broadcast_mode == "all":
        # جميع المستخدمين
        recipients = list(ALLOWED_USERS.union(DISALLOWED_USERS))
        target_name = get_text(lang, "send_message_target_all")
    elif broadcast_mode == "specific":
        # معالجة تنسيق معرف المستخدم:رسالة
        message_text = update.message.text
        
        if ':' not in message_text:
            await update.message.reply_text(
                get_text(lang, "message_format_error"),
                parse_mode='HTML',
                reply_markup=InlineKeyboardMarkup([
                    [InlineKeyboardButton(get_text(lang, "send_another_message"), callback_data="admin_send_to_specific")],
                    [InlineKeyboardButton(get_text(lang, "back"), callback_data="admin_panel")]
                ])
            )
            return
        
        parts = message_text.split(':', 1)
        target_user_id_str = parts[0].strip()
        user_message = parts[1].strip()
        
        # تحقق من أن المعرف رقم
        if not target_user_id_str.isdigit():
            await update.message.reply_text(
                get_text(lang, "invalid_user_id_format"),
                reply_markup=InlineKeyboardMarkup([
                    [InlineKeyboardButton(get_text(lang, "send_another_message"), callback_data="admin_send_to_specific")],
                    [InlineKeyboardButton(get_text(lang, "back"), callback_data="admin_panel")]
                ])
            )
            return
        
        # تحقق من وجود المستخدم في البوت
        if target_user_id_str not in ALLOWED_USERS and target_user_id_str not in DISALLOWED_USERS:
            await update.message.reply_text(
                get_text(lang, "user_not_found_in_bot"),
                reply_markup=InlineKeyboardMarkup([
                    [InlineKeyboardButton(get_text(lang, "send_another_message"), callback_data="admin_send_to_specific")],
                    [InlineKeyboardButton(get_text(lang, "back"), callback_data="admin_panel")]
                ])
            )
            return
        
        # إرسال الرسالة للمستخدم المستهدف
        try:
            await context.bot.send_message(int(target_user_id_str), user_message)
            await update.message.reply_text(
                get_text(lang, "message_sent_specific_user", user_id=target_user_id_str),
                reply_markup=InlineKeyboardMarkup([
                    [InlineKeyboardButton(get_text(lang, "send_another_message"), callback_data="admin_send_to_specific")],
                    [InlineKeyboardButton(get_text(lang, "back"), callback_data="admin_panel")]
                ])
            )
        except Exception as e:
            await update.message.reply_text(
                get_text(lang, "failed_send_specific_user", user_id=target_user_id_str, error=str(e)),
                reply_markup=InlineKeyboardMarkup([
                    [InlineKeyboardButton(get_text(lang, "send_another_message"), callback_data="admin_send_to_specific")],
                    [InlineKeyboardButton(get_text(lang, "back"), callback_data="admin_panel")]
                ])
            )
        
        # مسح وضع البث
        del BROADCAST_MODE[user_id]
        return
        
    elif broadcast_mode == "specific_user":
        # مستخدم محدد
        target_user_id = context.user_data.get('target_user_id')
        if not target_user_id:
            await update.message.reply_text(get_text(lang, "id_not_found_message"))
            del BROADCAST_MODE[user_id]
            return
        
        recipients = [target_user_id]
        target_name = f"مستخدم محدد (ID: {target_user_id})"
    else:
        await update.message.reply_text("❌ وضع البث غير معروف!")
        return
    
    # إذا كان الوضع ليس "specific"، نتعامل مع البث العادي
    if broadcast_mode != "specific":
        # إرسال رسالة التقدم
        progress_msg = await update.message.reply_text(f"⏳ جاري إرسال الرسالة إلى {target_name}...")
        
        sent_count = 0
        failed_count = 0
        
        # إرسال الرسالة لكل مستخدم
        for user_id_str in recipients:
            try:
                user_int_id = int(user_id_str)
                
                if update.message.text:
                    await context.bot.send_message(user_int_id, update.message.text)
                    sent_count += 1
                elif update.message.photo:
                    await context.bot.send_photo(user_int_id, update.message.photo[-1].file_id, caption=update.message.caption)
                    sent_count += 1
                elif update.message.video:
                    await context.bot.send_video(user_int_id, update.message.video.file_id, caption=update.message.caption)
                    sent_count += 1
                elif update.message.audio:
                    await context.bot.send_audio(user_int_id, update.message.audio.file_id, caption=update.message.caption)
                    sent_count += 1
                elif update.message.voice:
                    await context.bot.send_voice(user_int_id, update.message.voice.file_id, caption=update.message.caption)
                    sent_count += 1
                elif update.message.document:
                    await context.bot.send_document(user_int_id, update.message.document.file_id, caption=update.message.caption)
                    sent_count += 1
                
                # تأخير بين الرسائل لتجنب الحظر
                await asyncio.sleep(0.1)
                
            except Exception as e:
                print(f"❌ فشل في الإرسال إلى {user_id_str}: {e}")
                failed_count += 1
        
        # حذف رسالة التقدم
        try:
            await progress_msg.delete()
        except:
            pass
        
        # حذف رسالة المشرف الأصلية
        try:
            await update.message.delete()
        except:
            pass
        
        # إرسال تقرير النتيجة
        await context.bot.send_message(
            user_id,
            get_text(lang, "message_sent_success", count=sent_count),
            reply_markup=InlineKeyboardMarkup([
                [InlineKeyboardButton(get_text(lang, "send_another_message"), callback_data=f"admin_send_to_{broadcast_mode}")],
                [InlineKeyboardButton(get_text(lang, "back_to_panel"), callback_data="admin_panel")]
            ])
        )
    
    # مسح وضع البث
    if user_id in BROADCAST_MODE:
        del BROADCAST_MODE[user_id]
    if 'target_user_id' in context.user_data:
        del context.user_data['target_user_id']

        return

        return
    
    user_input = update.message.text.strip()
    
    # إرسال رسالة الانتظار
    wait_msg = await update.message.reply_text("🔍 جاري البحث عن المستخدم...")
    
    # البحث عن المستخدم في جميع المصادر
    target_user_id = None
    username = get_text("ar", "user_not_found")
    pocket_id = get_text("ar", "user_not_found")
    user_found = False
    
    # البحث في ALLOWED_USERS
    if user_input in ALLOWED_USERS:
        target_user_id = user_input
        user_found = True
    # البحث في DISALLOWED_USERS
    elif user_input in DISALLOWED_USERS:
        target_user_id = user_input
        user_found = True
    # البحث في PENDING_VERIFICATIONS
    elif user_input in PENDING_VERIFICATIONS:
        target_user_id = user_input
        user_found = True
    # البحث في POCKET_IDS بالقيمة
    else:
        for uid, pid in POCKET_IDS.items():
            if pid == user_input:
                target_user_id = uid
                pocket_id = pid
                user_found = True
                break
    
    await wait_msg.delete()
    
    if user_found and target_user_id:
        try:
            user = await context.bot.get_chat(int(target_user_id))
            username = f"@{user.username}" if user.username else user.first_name or get_text("ar", "user_not_found")
        except:
            username = get_text("ar", "user_not_found")
        
        # الحصول على pocket_id إذا لم يكن معروفاً
        if pocket_id == get_text("ar", "user_not_found"):
            pocket_id = POCKET_IDS.get(target_user_id, get_text("ar", "user_not_found"))
        
        text = get_text("ar", "user_found",
                       username=username,
                       user_id=target_user_id,
                       pocket_id=pocket_id)
        
        keyboard = [
            [
                InlineKeyboardButton(get_text("ar", "accept_user"), callback_data=f"acc_{target_user_id}"),
                InlineKeyboardButton(get_text("ar", "reject_user"), callback_data=f"rej_{target_user_id}")

        ]
        
        message = await update.message.reply_text(text, reply_markup=InlineKeyboardMarkup(keyboard))
        
    else:
        message = await update.message.reply_text(
            get_text("ar", "user_not_found"),
            reply_markup=InlineKeyboardMarkup([
                [InlineKeyboardButton(get_text("ar", "back"), callback_data="admin_panel")]
            ])
        )
    
    context.user_data.pop('awaiting_verify_id', None)

async def handle_pocket_id(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """معالجة إدخال الـ ID مع تحديث الصورة"""
    user_id = update.effective_user.id
    lang = context.user_data.get("lang", get_user_language(user_id))
    
    if not context.user_data.get('awaiting_pocket_id'):
        print(f"❌ User {user_id} is not in awaiting_pocket_id state")
        return
    
    pocket_id = update.message.text.strip()
    
    # حذف رسالة التعليمات أولاً
    if 'instruction_message_id' in context.user_data:
        try:
            await context.bot.delete_message(
                chat_id=update.effective_chat.id,
                message_id=context.user_data['instruction_message_id']
            )
            del context.user_data['instruction_message_id']
        except Exception as e:
            print(f"⚠️ Could not delete instruction message: {e}")
    
    # حذف رسالة المستخدم أيضاً
    try:
        await update.message.delete()
    except Exception as e:
        print(f"⚠️ Could not delete user message: {e}")
    
    # التحقق من صحة الـ ID - تم التعديل من 9 إلى 8 أرقام
    if not pocket_id.isdigit() or (len(pocket_id) != 8 and len(pocket_id) != 9):
        error_msg = await update.message.reply_text(
            get_text(lang, "id_length_error"),
            reply_markup=InlineKeyboardMarkup([
                [InlineKeyboardButton(get_text(lang, "technical_support"), url="https://t.me/H_DRJ")],
                [InlineKeyboardButton(get_text(lang, "back"), callback_data="verify_registration")]
            ])
        )
        return
    
    # التحقق من محاولات المستخدم
    user_id_str = str(user_id)
    today = datetime.datetime.now().strftime("%Y-%m-%d")
    
    if user_id_str not in USER_ATTEMPTS:
        USER_ATTEMPTS[user_id_str] = {}
    
    if today not in USER_ATTEMPTS[user_id_str]:
        USER_ATTEMPTS[user_id_str][today] = 0
    
    # التحقق من عدد المحاولات
    if USER_ATTEMPTS[user_id_str][today] >= 3:
        error_msg = await update.message.reply_text(get_text(lang, "id_max_attempts"))
        return
    
    # التحقق إذا كان الـ ID مستخدم من قبل لنفس المستخدم
    if user_id_str in POCKET_IDS and POCKET_IDS[user_id_str] == pocket_id:
        error_msg = await update.message.reply_text(get_text(lang, "id_same_used"))
        return
    
    # التحقق إذا كان الـ ID مستخدم من قبل لمستخدم آخر
    if pocket_id in POCKET_IDS.values():
        error_msg = await update.message.reply_text(get_text(lang, "id_already_used"))
        return
    
    # زيادة عدد المحاولات
    USER_ATTEMPTS[user_id_str][today] += 1
    save_json_dict(USER_ATTEMPTS_FILE, USER_ATTEMPTS)
    
    # حفظ الـ ID
    POCKET_IDS[user_id_str] = pocket_id
    save_json_dict(POCKET_IDS_FILE, POCKET_IDS)
    
    # إضافة للتحقق المعلق
    PENDING_VERIFICATIONS[user_id_str] = {
        'username': update.effective_user.username or update.effective_user.first_name,
        'pocket_id': pocket_id,
        'timestamp': datetime.datetime.now().isoformat()
    }
    save_json_dict(PENDING_VERIFICATIONS_FILE, PENDING_VERIFICATIONS)
    
    context.user_data.pop('awaiting_pocket_id', None)
    
    # عملية التحقق مع الرسائل المتتابعة
    progress_message = await update.message.reply_text(get_text(lang, "checking_steps")[0])
    
    steps = get_text(lang, "checking_steps")
    for i in range(1, len(steps)):
        await asyncio.sleep(1)
        try:
            await progress_message.edit_text(steps[i])
        except:
            pass
    
    await asyncio.sleep(1)
    await progress_message.delete()
    
    # إرسال رسالة الانتظار النهائية مع الصورة الجديدة
    final_msg = await update.message.reply_photo(
        photo="https://i.postimg.cc/zGg5FSX0/20251021-070056.jpg",
        caption=get_text(lang, "final_check_message"),
        reply_markup=InlineKeyboardMarkup([
            [InlineKeyboardButton(get_text(lang, "back"), callback_data="back_to_main")]
        ])
    )
    
    # إرسال إشعار للمالك فقط
    try:
        await context.bot.send_message(
            OWNER_ID,
            get_text("ar", "new_verification",
                    username=update.effective_user.username or update.effective_user.first_name,
                    user_id=user_id,
                    pocket_id=pocket_id),
            reply_markup=InlineKeyboardMarkup([
                [
                    InlineKeyboardButton(get_text("ar", "accept_user"), callback_data=f"acc_{user_id_str}"),
                    InlineKeyboardButton(get_text("ar", "reject_user"), callback_data=f"rej_{user_id_str}")
                ]
            ])
        )
    except Exception as e:
        print(f"Failed to send notification to owner: {e}")

async def allow_user(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """منح صلاحية استخدام البوت - النسخة المحسنة"""
    user_id = update.effective_user.id
    lang = context.user_data.get("lang", get_user_language(user_id))

    if not user_is_admin(user_id):
        await update.message.reply_text(get_text(lang, "no_permission"))
        return

    if not context.args:
        await update.message.reply_text(get_text(lang, "provide_user_id"))
        return

    try:
        new_user_id = int(context.args[0])
    except ValueError:
        await update.message.reply_text(get_text(lang, "invalid_user_id"))
        return

async def backup_data(context: ContextTypes.DEFAULT_TYPE):
    """إنشاء نسخة احتياطية كل ساعة"""
    try:
        timestamp = int(time.time())
        backup_dir = "backups"
        if not os.path.exists(backup_dir):
            os.makedirs(backup_dir)
        
        # نسخ جميع الملفات المهمة
        important_files = [
            ALLOWED_USERS_FILE, DISALLOWED_USERS_FILE, BOT_STATS_FILE,
            USER_HISTORIES_FILE, REFERRALS_FILE, POCKET_IDS_FILE,
            PENDING_VERIFICATIONS_FILE, USER_ATTEMPTS_FILE,
            DEPOSIT_CONFIRMED_FILE, SUBSCRIPTIONS_FILE,
            SUBSCRIPTION_MESSAGES_FILE, NEW_USERS_LOG_FILE,
            USER_LANGUAGES_FILE
        ]
        
        for file in important_files:
            if os.path.exists(file):
                backup_file = os.path.join(backup_dir, f"{file}.backup_{timestamp}")
                with open(file, 'r', encoding='utf-8') as src, open(backup_file, 'w', encoding='utf-8') as dst:
                    dst.write(src.read())
        
        print(f"📦 تم إنشاء نسخة احتياطية: {timestamp}")
        
    except Exception as e:
        print(f"❌ خطأ في النسخ الاحتياطي: {e}")

async def handle_subscription_user_id(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """معالجة معرف المستخدم لإنشاء اشتراك"""
    user_id = update.effective_user.id
    
    if not user_is_admin(user_id) or not context.user_data.get('awaiting_subscription_user_id'):
        return
    
    user_input = update.message.text.strip()
    
    # تحقق من التنسيق
    parts = user_input.split()
    if len(parts) != 2:
        await update.message.reply_text(
            "❌ تنسيق غير صحيح!\n\n"
            "يجب أن يكون: <code>معرف_المستخدم عدد_الأيام</code>\n"
            "مثال: 123456789 30",
            parse_mode='HTML'
        )
        return
    
    target_user_id_str, days_str = parts[0].strip(), parts[1].strip()
    
    # تحقق من المعرف
    if not target_user_id_str.isdigit():
        await update.message.reply_text("❌ معرف المستخدم يجب أن يكون رقماً!")
        return
    
    # تحقق من عدد الأيام
    try:
        days = int(days_str)
        if days <= 0:
            await update.message.reply_text("❌ عدد الأيام يجب أن يكون أكبر من الصفر!")
            return
    except ValueError:
        await update.message.reply_text("❌ عدد الأيام يجب أن يكون رقماً!")
        return
    
    # حفظ البيانات المؤقتة
    context.user_data['subscription_data'] = {
        'target_user_id': target_user_id_str,
        'days': days
    }
    context.user_data['awaiting_subscription_user_id'] = False
    context.user_data['awaiting_subscription_message'] = True
    
    # طلب رسالة الاشتراك
    lang = context.user_data.get("lang", get_user_language(user_id))
    await update.message.reply_text(
        get_text(lang, "subscription_data_received", user_id=target_user_id_str, days=days),
        reply_markup=InlineKeyboardMarkup([
            [InlineKeyboardButton("❌ إلغاء", callback_data="cancel_subscription")]
        ])
    )

async def handle_subscription_message(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """معالجة رسالة الاشتراك وإكمال العملية"""
    user_id = update.effective_user.id
    
    if not user_is_admin(user_id) or not context.user_data.get('awaiting_subscription_message'):
        return
    
    if 'subscription_data' not in context.user_data:
        await update.message.reply_text("❌ خطأ: بيانات الاشتراك غير موجودة!")
        return
    
    subscription_data = context.user_data['subscription_data']
    target_user_id_str = subscription_data['target_user_id']
    days = subscription_data['days']
    admin_message = update.message.text
    
    # حساب تواريخ البدء والانتهاء
    start_date = datetime.datetime.now()
    end_date = start_date + datetime.timedelta(days=days)
    
    # حفظ بيانات الاشتراك
    SUBSCRIPTIONS[target_user_id_str] = {
        'start_date': start_date.isoformat(),
        'end_date': end_date.isoformat(),
        'days': days,
        'added_by': user_id,
        'added_at': datetime.datetime.now().isoformat()
    }
    
    # حفظ رسالة الاشتراك
    SUBSCRIPTION_MESSAGES[target_user_id_str] = admin_message
    
    # منح الصلاحية للمستخدم (إضافة إلى المسموح لهم)
    ALLOWED_USERS.add(target_user_id_str)
    if target_user_id_str in DISALLOWED_USERS:
        DISALLOWED_USERS.remove(target_user_id_str)
    
    # حفظ البيانات
    save_json_dict(SUBSCRIPTIONS_FILE, SUBSCRIPTIONS)
    save_json_dict(SUBSCRIPTION_MESSAGES_FILE, SUBSCRIPTION_MESSAGES)
    save_json_set(ALLOWED_USERS_FILE, ALLOWED_USERS)
    save_json_set(DISALLOWED_USERS_FILE, DISALLOWED_USERS)
    
    # مسح البيانات المؤقتة
    context.user_data.pop('subscription_data', None)
    context.user_data.pop('awaiting_subscription_message', None)
    
    # إرسال الرسالة للمستخدم
    try:
        user_lang = get_user_language(int(target_user_id_str))
        await context.bot.send_photo(
            int(target_user_id_str),
            photo="https://i.postimg.cc/W17G1yQh/20251021-085511.jpg",
            caption=f"{admin_message}\n\n⏰ {get_text(user_lang, 'days_left').format(days)}\n📅 {get_text(user_lang, 'subscription_until').format(end_date.strftime('%Y-%m-%d %H:%M'))}",
            reply_markup=InlineKeyboardMarkup([
                [InlineKeyboardButton(get_text(user_lang, "renew_subscription"), url="https://t.me/AIVEX_ACCESS_BOT")],
                [InlineKeyboardButton(get_text(user_lang, "get_signal"), callback_data="get_signal")]
            ])
        )
        print(f"✅ تم إرسال رسالة الاشتراك للمستخدم {target_user_id_str}")
    except Exception as e:
        print(f"❌ فشل في إرسال الرسالة للمستخدم {target_user_id_str}: {e}")
    
    lang = context.user_data.get("lang", get_user_language(user_id))
    await update.message.reply_text(
        get_text(lang, "subscription_activated", user_id=target_user_id_str, days=days, end_date=end_date.strftime('%Y-%m-%d %H:%M')),
        reply_markup=InlineKeyboardMarkup([
            [InlineKeyboardButton(get_text(lang, "add_subscription"), callback_data="add_subscription_command")],
            [InlineKeyboardButton(get_text(lang, "back_to_panel"), callback_data="admin_panel")]
        ])
    )

async def cancel_subscription_handler(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """إلغاء عملية إضافة اشتراك"""
    user_id = update.effective_user.id
    q = update.callback_query

        ])
    )

async def admin_subscriptions_handler(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """عرض المشتركين في لوحة المشرفين - النسخة المحسنة"""
    user_id = update.effective_user.id
    q = update.callback_query
    await q.answer()
    lang = context.user_data.get("lang", get_user_language(user_id))
    
    if not user_is_admin(user_id):
        return
    
    try:
        await q.message.delete()
    except Exception:
        pass
    
    # استخدام النصوص المترجمة
    title_text = get_text(lang, "subscriptions_list")
    active_text = get_text(lang, "subscription_active")
    expired_text = get_text(lang, "subscription_expired")
    days_left_text = get_text(lang, "days_left")
    until_text = get_text(lang, "subscription_until")
    refresh_text = get_text(lang, "refresh_stats")
    back_text = get_text(lang, "back_to_panel")
    add_text = get_text(lang, "add_subscription")
    
    text = f"👥 **{title_text}**\n\n"
    
    if not SUBSCRIPTIONS:
        text += get_text(lang, "no_subscribers")
    else:
        current_time = datetime.datetime.now()
        active_count = 0
        expired_count = 0
        
        for user_id_str, sub_data in SUBSCRIPTIONS.items():
            try:
                end_date = datetime.datetime.fromisoformat(sub_data['end_date'])
                days_left = (end_date - current_time).days
                
                # الحصول على اسم المستخدم
                try:
                    user = await context.bot.get_chat(int(user_id_str))
                    username = f"@{user.username}" if user.username else user.first_name or "لا يوجد معرف"
                except:
                    username = "مستخدم"
                
                if days_left > 0:
                    active_count += 1
                    status = active_text
                    days_text = days_left_text.format(days_left)
                    until_text_formatted = until_text.format(end_date.strftime('%Y-%m-%d'))
                    
                    text += f"👤 {username}\n🆔: {user_id_str}\n{days_text}\n{until_text_formatted}\n{status}\n────────────────────\n"
                else:
                    expired_count += 1
                
            except Exception as e:
                print(f"Error processing subscription for {user_id_str}: {e}")
                continue
        
        # إضافة ملخص
        text += f"\n{get_text(lang, 'subscription_summary').format(active_count=active_count, expired_count=expired_count, total_count=active_count + expired_count)}"
    
    keyboard = [
        [InlineKeyboardButton(refresh_text, callback_data="admin_subscriptions")],
        [InlineKeyboardButton(add_text, callback_data="add_subscription_command")],
        [InlineKeyboardButton(back_text, callback_data="admin_panel")]
    ]
    
    message = await q.message.chat.send_message(text, reply_markup=InlineKeyboardMarkup(keyboard))

async def check_subscription_expiry(context: ContextTypes.DEFAULT_TYPE):
    """فحص انتهاء صلاحية الاشتراكات وإرسال تنبيهات - النسخة المحسنة"""
    try:
        current_time = datetime.datetime.now()
        
        # التحقق من أن اليوم هو الجمعة والساعة 12 ظهراً
        if current_time.weekday() == 4 and current_time.hour == 12 and current_time.minute == 0:
            for user_id_str, sub_data in SUBSCRIPTIONS.items():
                try:
                    end_date = datetime.datetime.fromisoformat(sub_data['end_date'])
                    days_left = (end_date - current_time).days
                    
                    # الحصول على لغة المستخدم المحفوظة
                    user_lang = get_user_language(int(user_id_str)) if user_id_str.isdigit() else "ar"
                    
                    # إرسال رسالة تنبيه إذا انتهى الاشتراك
                    if days_left <= 0:
                        try:
                            await context.bot.send_message(
                                int(user_id_str),
                                get_text(user_lang, "subscription_expired_message"),
                                reply_markup=InlineKeyboardMarkup([
                                    [InlineKeyboardButton(get_text(user_lang, "renew_subscription"), url="https://t.me/AIVEX_ACCESS_BOT")],
                                    [InlineKeyboardButton(get_text(user_lang, "technical_support"), url="https://t.me/H_DRJ")]
                                ])
                            )
                            print(f"✅ تم إرسال رسالة انتهاء الاشتراك للمستخدم {user_id_str}")
                        except Exception as e:
                            print(f"❌ فشل في إرسال رسالة انتهاء الاشتراك: {e}")
                    
                except Exception as e:
                    print(f"❌ خطأ في معالجة اشتراك المستخدم {user_id_str}: {e}")
                    continue
        
        print(f"✅ تم فحص انتهاء الاشتراكات في {current_time.strftime('%Y-%m-%d %H:%M')}")
        
    except Exception as e:
        print(f"❌ خطأ في فحص انتهاء الاشتراكات: {e}")
